132 template <typename ValueT>
133 std::pair<bool, ValueT> fromSettingString(const std::string& str, const EnumOrString<ValueT>& defaultValue) {
134  if (str.length() < 2 || str[0] != QUOTE || str[str.length() - 1] != QUOTE) {
135  return {false, defaultValue};

137 
138  ValueT value = defaultValue;
