212 template <typename T>
213 SharedDataStream<T>::Reader::Reader(Policy policy, std::shared_ptr<BufferLayout> bufferLayout, uint8_t id) :
214  m_policy{policy},
215  m_bufferLayout{bufferLayout},
216  m_id{id},
217  m_readerCursor{&m_bufferLayout->getReaderCursorArray()[m_id]},
218  m_readerCloseIndex{&m_bufferLayout->getReaderCloseIndexArray()[m_id]} {
219  // Note - SharedDataStream::createReader() holds readerEnableMutex while calling this function.
220  // Read new data only.
221  // Note: It is important that new readers start with their cursor at the writer. This allows
222  // updateOldestUnconsumedCursor() to be thread-safe without holding readerEnableMutex. See
223  // updateOldestUnconsumedCursor() comments for further explanation.
224  *m_readerCursor = m_bufferLayout->getHeader()->writeStartCursor.load();
225 
226  // Read indefinitely.
227  *m_readerCloseIndex = std::numeric_limits<Index>::max();
228 
229  m_bufferLayout->enableReaderLocked(m_id);
230 }

232 template <typename T>
233 SharedDataStream<T>::Reader::~Reader() {
234  // Note: We can't leave a reader with its cursor in the future; doing so can introduce a race condition in
235  // updateOldestUnconsumedCursor(). See updateOldestUnconsumedCursor() comments for further explanation.
236  seek(0, Reference::BEFORE_WRITER);
237 
238  std::lock_guard<Mutex> lock(m_bufferLayout->getHeader()->readerEnableMutex);
239  m_bufferLayout->disableReaderLocked(m_id);
240  m_bufferLayout->updateOldestUnconsumedCursor();
241 }

243 template <typename T>
244 ssize_t SharedDataStream<T>::Reader::read(void* buf, size_t nWords, std::chrono::milliseconds timeout) {
245  if (nullptr == buf) {
246  logger::acsdkError(logger::LogEntry(TAG, "readFailed").d("reason", "nullBuffer"));
247  return Error::INVALID;
248  }
249 
250  if (0 == nWords) {
251  logger::acsdkError(logger::LogEntry(TAG, "readFailed").d("reason", "invalidNumWords").d("numWords", nWords));
252  return Error::INVALID;
253  }
254 
255  // Check if closed.
256  auto readerCloseIndex = m_readerCloseIndex->load();
257  if (*m_readerCursor >= readerCloseIndex) {
258  return Error::CLOSED;
259  }
260 
261  // Initial check for overrun.
262  auto header = m_bufferLayout->getHeader();
263  if ((header->writeEndCursor >= *m_readerCursor) &&
264  (header->writeEndCursor - *m_readerCursor) > m_bufferLayout->getDataSize()) {
265  return Error::OVERRUN;
266  }
267 
268  std::unique_lock<Mutex> lock(header->dataAvailableMutex, std::defer_lock);
269  if (Policy::BLOCKING == m_policy) {
270  lock.lock();
271  }
272 
273  // Figure out how much we can actually copy.
274  size_t wordsAvailable = tell(Reference::BEFORE_WRITER);
275  if (0 == wordsAvailable) {
276  if (header->writeEndCursor > 0 && !header->isWriterEnabled) {
277  return Error::CLOSED;
278  } else if (Policy::NONBLOCKING == m_policy) {
279  return Error::WOULDBLOCK;
280  } else if (Policy::BLOCKING == m_policy) {
281  // Condition for returning from read: the Writer has been closed or there is data to read
282  auto predicate = [this, header] {
283  return header->hasWriterBeenClosed || tell(Reference::BEFORE_WRITER) > 0;
284  };
285 
286  if (std::chrono::milliseconds::zero() == timeout) {
287  header->dataAvailableConditionVariable.wait(lock, predicate);
288  } else if (!header->dataAvailableConditionVariable.wait_for(lock, timeout, predicate)) {
289  return Error::TIMEDOUT;
290  }
291  }
292  wordsAvailable = tell(Reference::BEFORE_WRITER);
293 
294  // If there is still no data, the writer has closed in the interim
295  if (0 == wordsAvailable) {
296  return Error::CLOSED;
297  }
298  }
299 
300  if (Policy::BLOCKING == m_policy) {
301  lock.unlock();
302  }
303  if (nWords > wordsAvailable) {
304  nWords = wordsAvailable;
305  }
306 
307  // Don't read beyond our close index.
308  if ((*m_readerCursor + nWords) > readerCloseIndex) {
309  nWords = readerCloseIndex - *m_readerCursor;
310  }
311 
312  // Split it across the wrap.
313  size_t beforeWrap = m_bufferLayout->wordsUntilWrap(*m_readerCursor);
314  if (beforeWrap > nWords) {
315  beforeWrap = nWords;
316  }
317  size_t afterWrap = nWords - beforeWrap;
318 
319  // Copy the two segments.
320  auto buf8 = static_cast<uint8_t*>(buf);
321  memcpy(buf8, m_bufferLayout->getData(*m_readerCursor), beforeWrap * getWordSize());
322  if (afterWrap > 0) {
323  memcpy(
324  buf8 + (beforeWrap * getWordSize()),
325  m_bufferLayout->getData(*m_readerCursor + beforeWrap),
326  afterWrap * getWordSize());
327  }
328 
329  // Advance the read cursor.
330  *m_readerCursor += nWords;
331 
332  // Final check for overrun (do this before the updateOldestUnconsumedCursor() call below for improved accuracy).
333  bool overrun = ((header->writeEndCursor - *m_readerCursor) > m_bufferLayout->getDataSize());
334 
335  // Move the unconsumed cursor before returning.
336  m_bufferLayout->updateOldestUnconsumedCursor();
337 
338  // Now we can safely error out if there was an overrun.
339  if (overrun) {
340  return Error::OVERRUN;
341  }
342 
343  return nWords;
344 }

346 template <typename T>
347 bool SharedDataStream<T>::Reader::seek(Index offset, Reference reference) {
348  auto header = m_bufferLayout->getHeader();
349  auto writeStartCursor = &header->writeStartCursor;
350  auto writeEndCursor = &header->writeEndCursor;
351  Index absolute = std::numeric_limits<Index>::max();
352 
353  switch (reference) {
354  case Reference::AFTER_READER:
355  absolute = *m_readerCursor + offset;
356  break;
357  case Reference::BEFORE_READER:
358  if (offset > *m_readerCursor) {
359  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
360  .d("reason", "seekBeforeStreamStart")
361  .d("reference", "BEFORE_READER")
362  .d("seekOffset", offset)
363  .d("readerCursor", m_readerCursor->load()));
364  return false;
365  }
366  absolute = *m_readerCursor - offset;
367  break;
368  case Reference::BEFORE_WRITER:
369  if (offset > *writeStartCursor) {
370  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
371  .d("reason", "seekBeforeStreamStart")
372  .d("reference", "BEFORE_WRITER")
373  .d("seekOffset", offset)
374  .d("writeStartCursor", writeStartCursor->load()));
375  return false;
376  }
377  absolute = *writeStartCursor - offset;
378  break;
379  case Reference::ABSOLUTE:
380  absolute = offset;
381  }
382 
383  // Don't seek beyond the close index.
384  if (absolute > *m_readerCloseIndex) {
385  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
386  .d("reason", "seekBeyondCloseIndex")
387  .d("position", absolute)
388  .d("readerCloseIndex", m_readerCloseIndex->load()));
389  return false;
390  }
391 
392  // Per documentation of updateOldestUnconsumedCursor(), don't try to seek backwards while oldestConsumedCursor is
393  // being updated.
394  bool backward = absolute < *m_readerCursor;
395  std::unique_lock<Mutex> lock(header->backwardSeekMutex, std::defer_lock);
396  if (backward) {
397  lock.lock();
398  }
399 
400  // Don't seek to past data which has been (or soon will be) overwritten.
401  // Note: If this is a backward seek, it is important that this check is performed while holding the
402  // backwardSeekMutex to prevent a writer from starting to overwrite us between here and the m_readerCursor update
403  // below. If this is not a backward seek, then the mutex is not held.
404  if (*writeEndCursor >= absolute && *writeEndCursor - absolute > m_bufferLayout->getDataSize()) {
405  logger::acsdkError(logger::LogEntry(TAG, "seekFailed").d("reason", "seekOverwrittenData"));
406  return false;
407  }
408 
409  *m_readerCursor = absolute;
410 
411  if (backward) {
412  m_bufferLayout->updateOldestUnconsumedCursorLocked();
413  lock.unlock();
414  } else {
415  m_bufferLayout->updateOldestUnconsumedCursor();
416  }
417 
418  return true;
419 }

421 template <typename T>
422 typename SharedDataStream<T>::Index SharedDataStream<T>::Reader::tell(Reference reference) const {
423  auto writeStartCursor = &m_bufferLayout->getHeader()->writeStartCursor;
424  switch (reference) {
425  case Reference::AFTER_READER:
426  return 0;
427  case Reference::BEFORE_READER:
428  return 0;
429  case Reference::BEFORE_WRITER:
430  return (*writeStartCursor >= *m_readerCursor) ? *writeStartCursor - *m_readerCursor : 0;
431  case Reference::ABSOLUTE:
432  return *m_readerCursor;
433  }
434  logger::acsdkError(logger::LogEntry(TAG, "tellFailed").d("reason", "invalidReference"));
435  return std::numeric_limits<Index>::max();
436 }

438 template <typename T>
439 void SharedDataStream<T>::Reader::close(Index offset, Reference reference) {
440  auto writeStartCursor = &m_bufferLayout->getHeader()->writeStartCursor;
441  Index absolute = 0;
442  bool validReference = false;
443  switch (reference) {
444  case Reference::AFTER_READER:
445  absolute = *m_readerCursor + offset;
446  validReference = true;
447  break;
448  case Reference::BEFORE_READER:
449  absolute = *m_readerCursor;
450  validReference = true;
451  break;
452  case Reference::BEFORE_WRITER:
453  if (*writeStartCursor < offset) {
454  logger::acsdkError(logger::LogEntry(TAG, "closeFailed")
455  .d("reason", "invalidIndex")
456  .d("reference", "BEFORE_WRITER")
457  .d("offset", offset)
458  .d("writeStartCursor", writeStartCursor->load()));
459  } else {
460  absolute = *writeStartCursor - offset;
461  }
462  validReference = true;
463  break;
464  case Reference::ABSOLUTE:
465  absolute = offset;
466  validReference = true;
467  break;
468  }
469  if (!validReference) {
470  logger::acsdkError(logger::LogEntry(TAG, "closeFailed").d("reason", "invalidReference"));
471  }
472 
473  *m_readerCloseIndex = absolute;
474 }
