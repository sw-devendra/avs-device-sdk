309 template <typename T>
310 size_t SharedDataStream<T>::calculateBufferSize(size_t nWords, size_t wordSize, size_t maxReaders) {
311  if (0 == nWords) {
312  logger::acsdkError(logger::LogEntry(TAG, "calculateBufferSizeFailed").d("reason", "numWordsZero"));
313  return 0;
314  } else if (0 == wordSize) {
315  logger::acsdkError(logger::LogEntry(TAG, "calculateBufferSizeFailed").d("reason", "wordSizeZero"));
316  return 0;
317  }
318  size_t overhead = BufferLayout::calculateDataOffset(wordSize, maxReaders);
319  size_t dataSize = nWords * wordSize;
320  return overhead + dataSize;
321 }

379  typename Writer::Policy policy,
380  bool forceReplacement) {
381  auto header = m_bufferLayout->getHeader();
382  std::lock_guard<Mutex> lock(header->writerEnableMutex);
383  if (header->isWriterEnabled && !forceReplacement) {
384  logger::acsdkError(logger::LogEntry(TAG, "createWriterFailed")
385  .d("reason", "existingWriterAttached")
386  .d("forceReplacement", "false"));
387  return nullptr;
388  } else {
389  return std::unique_ptr<Writer>(new Writer(policy, m_bufferLayout));

393 template <typename T>
394 std::unique_ptr<typename SharedDataStream<T>::Reader> SharedDataStream<T>::createReader(
395  typename Reader::Policy policy,
396  bool startWithNewData) {
397  std::unique_lock<Mutex> lock(m_bufferLayout->getHeader()->readerEnableMutex);
398  for (size_t id = 0; id < m_bufferLayout->getHeader()->maxReaders; ++id) {
399  if (!m_bufferLayout->isReaderEnabled(id)) {
400  return createReaderLocked(id, policy, startWithNewData, false, &lock);
401  }
402  }
403  logger::acsdkError(logger::LogEntry(TAG, "createWriterFailed").d("reason", "noAvailableReaders"));
404  return nullptr;
405 }

422 template <typename T>
423 std::unique_ptr<typename SharedDataStream<T>::Reader> SharedDataStream<T>::createReaderLocked(
424  size_t id,
425  typename Reader::Policy policy,
426  bool startWithNewData,
427  bool forceReplacement,
428  std::unique_lock<Mutex>* lock) {
429  if (m_bufferLayout->isReaderEnabled(id) && !forceReplacement) {
430  logger::acsdkError(logger::LogEntry(TAG, "createReaderLockedFailed")
431  .d("reason", "readerAlreadyAttached")
432  .d("readerId", id)
433  .d("forceReplacement", "false"));
434  return nullptr;
435  } else {
436  // Note: Reader constructor does not call updateUnconsumedCursor() automatically, because we may be seeking to
437  // a blocked writer's cursor below (if !startWithNewData), and we don't want the writer to start moving before
438  // we seek.
439  auto reader = std::unique_ptr<Reader>(new Reader(policy, m_bufferLayout, id));
440  lock->unlock();
441 
442  if (startWithNewData) {
443  // We're not moving the cursor again, so call updateUnconsumedCursor() now.
444  m_bufferLayout->updateOldestUnconsumedCursor();
445  } else {
446  Index offset = m_bufferLayout->getDataSize();
447  if (m_bufferLayout->getHeader()->writeStartCursor < offset) {
448  offset = m_bufferLayout->getHeader()->writeStartCursor;
449  }
450  // Note: seek() will call updateUnconsumedCursor().
451  if (!reader->seek(offset, Reader::Reference::BEFORE_WRITER)) {
452  // Logged in seek().
453  return nullptr;
454  }
455  }
456  return reader;
457  }
