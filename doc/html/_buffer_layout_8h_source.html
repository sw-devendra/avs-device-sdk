464 template <typename T>
465 bool SharedDataStream<T>::BufferLayout::init(size_t wordSize, size_t maxReaders) {
466  // Make sure parameters are not too large to store.
467  if (wordSize > std::numeric_limits<decltype(Header::wordSize)>::max()) {
468  logger::acsdkError(logger::LogEntry(TAG, "initFailed")
469  .d("reason", "wordSizeTooLarge")
470  .d("wordSize", wordSize)
471  .d("wordSizeLimit", std::numeric_limits<decltype(Header::wordSize)>::max()));
472  return false;
473  }
474  if (maxReaders > std::numeric_limits<decltype(Header::maxReaders)>::max()) {
475  logger::acsdkError(logger::LogEntry(TAG, "initFailed")
476  .d("reason", "maxReadersTooLarge")
477  .d("maxReaders", maxReaders)
478  .d("maxReadersLimit", std::numeric_limits<decltype(Header::maxReaders)>::max()));
479  return false;
480  }
481 
482  // Pre-calculate some pointers and sizes that are frequently accessed.
483  calculateAndCacheConstants(wordSize, maxReaders);
484 
485  // Default construction of the Header.
486  auto header = new (getHeader()) Header;
487 
488  // Default construction of the reader arrays.
489  size_t id;
490  for (id = 0; id < maxReaders; ++id) {
491  new (m_readerEnabledArray + id) AtomicBool;
492  new (m_readerCursorArray + id) AtomicIndex;
493  new (m_readerCloseIndexArray + id) AtomicIndex;
494  }
495 
496  // Header field initialization.
497  header->magic = MAGIC_NUMBER;
498  header->version = VERSION;
499  header->traitsNameHash = stableHash(T::traitsName);
500  header->wordSize = wordSize;
501  header->maxReaders = maxReaders;
502  header->isWriterEnabled = false;
503  header->hasWriterBeenClosed = false;
504  header->writeStartCursor = 0;
505  header->writeEndCursor = 0;
506  header->oldestUnconsumedCursor = 0;
507  header->referenceCount = 1;
508 
509  // Reader arrays initialization.
510  for (id = 0; id < maxReaders; ++id) {
511  m_readerEnabledArray[id] = false;
512  m_readerCursorArray[id] = 0;
513  m_readerCloseIndexArray[id] = 0;
514  }
515 
516  return true;
517 }

519 template <typename T>
520 bool SharedDataStream<T>::BufferLayout::attach() {
521  // Verify compatibility.
522  auto header = getHeader();
523  if (header->magic != MAGIC_NUMBER) {
524  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
525  .d("reason", "magicNumberMismatch")
526  .d("magicNumber", header->magic)
527  .d("expectedMagicNumber", std::to_string(MAGIC_NUMBER)));
528  return false;
529  }
530  if (header->version != VERSION) {
531  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
532  .d("reason", "incompatibleVersion")
533  .d("version", header->version)
534  .d("expectedVersion", std::to_string(VERSION)));
535  return false;
536  }
537  if (header->traitsNameHash != stableHash(T::traitsName)) {
538  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
539  .d("reason", "traitsNameHashMismatch")
540  .d("hash", header->traitsNameHash)
541  .d("expectedHash", stableHash(T::traitsName)));
542  return false;
543  }
544 
545  // Attach.
546  std::lock_guard<Mutex> lock(header->attachMutex);
547  if (0 == header->referenceCount) {
548  logger::acsdkError(logger::LogEntry(TAG, "attachFailed").d("reason", "zeroUsers"));
549  return false;
550  }
551  if (std::numeric_limits<decltype(header->referenceCount)>::max() == header->referenceCount) {
552  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
553  .d("reason", "bufferMaxUsersExceeded")
554  .d("numUsers", header->referenceCount)
555  .d("maxNumUsers", std::numeric_limits<decltype(header->referenceCount)>::max()));
556  return false;
557  }
558  ++header->referenceCount;
559 
560  // Pre-calculate some pointers and sizes that are frequently accessed.
561  calculateAndCacheConstants(header->wordSize, header->maxReaders);
562 
563  return true;
564 }

566 template <typename T>
567 void SharedDataStream<T>::BufferLayout::detach() {
568  if (!isAttached()) {
569  return;
570  }
571 
572  auto header = getHeader();
573  std::lock_guard<Mutex> lock(header->attachMutex);
574  --header->referenceCount;
575  if (header->referenceCount > 0) {
576  return;
577  }
578 
579  // Destruction of reader arrays.
580  for (size_t id = 0; id < header->maxReaders; ++id) {
581  m_readerCloseIndexArray[id].~AtomicIndex();
582  m_readerCursorArray[id].~AtomicIndex();
583  m_readerEnabledArray[id].~AtomicBool();
584  }
585 
586  // Destruction of the Header.
587  header->~Header();
588 }

622 template <typename T>
623 void SharedDataStream<T>::BufferLayout::updateOldestUnconsumedCursorLocked() {
624  auto header = getHeader();
625 
626  // Note: as an optimization, we could skip this function if Writer policy is nonblockable (ACSDK-251).
627 
628  // The only barrier to a blocking writer overrunning a reader is oldestUnconsumedCursor, so we have to be careful
629  // not to ever move it ahead of any readers. The loop below searches through the readers to find the oldest point,
630  // without moving oldestUnconsumedCursor. Note that readers can continue to read while we are looping; it means
631  // oldest may not be completely accurate, but it will always be older than the readers because they are reading
632  // away from it. Also note that backwards seeks (which would break the invariant) are prevented with a mutex which
633  // is held while this function is called. Also note that all read cursors may be in the future, so we start with
634  // an unlimited barrier and work back from there.
635  Index oldest = std::numeric_limits<Index>::max();
636  for (size_t id = 0; id < header->maxReaders; ++id) {
637  // Note that this code is calling isReaderEnabled() without holding readerEnableMutex. On the surface, this
638  // appears to be a race condition because a reader may be disabled and/or re-enabled before the subsequent code
639  // reads the cursor, but it turns out to be safe because:
640  // - if a reader is enabled, its cursor is valid
641  // - if a reader becomes disabled, its cursor moves to writeCursor (which will never be the oldest)
642  // - if a reader becomes re-enabled, its cursor defaults to writeCursor (which will never be the oldest)
643  // - if a reader is created that wants to be at an older index, it gets there by doing a backward seek (which
644  // is locked when this function is called)
645  if (isReaderEnabled(id) && getReaderCursorArray()[id] < oldest) {
646  oldest = getReaderCursorArray()[id];
647  }
648  }
649 
650  // If no barrier was found, block at the write cursor so that we retain data until a reader comes along to read it.
651  if (std::numeric_limits<Index>::max() == oldest) {
652  oldest = header->writeStartCursor;
653  }
654 
655  // Now that we've measured the oldest cursor, we can safely update oldestUnconsumedCursor with no risk of an
656  // overrun of any readers.
657 
658  // To clarify the logic here, the code above reviewed all of the enabled readers to see where the oldest cursor is
659  // at. This value is captured in the 'oldest' variable. Now we want to move up our writer barrier
660  // ('oldestUnconsumedCursor') if it is older than it needs to be.
661  if (oldest > header->oldestUnconsumedCursor) {
662  header->oldestUnconsumedCursor = oldest;
663 
664  // Notify the writer(s).
665  // Note: as an optimization, we could skip this if there are no blocking writers (ACSDK-251).
666  header->spaceAvailableConditionVariable.notify_all();
667  }
668 }

670 template <typename T>
671 uint32_t SharedDataStream<T>::BufferLayout::stableHash(const char* string) {
672  // Simple, stable hash which XORs all bytes of string into the hash value.
673  uint32_t hashed = 0;
674  size_t pos = 0;
675  while (*string) {
676  hashed ^= *string << ((pos % sizeof(uint32_t)) * 8);
677  ++string;
678  ++pos;
679  }
680  return hashed;
681 }

707 template <typename T>
708 void SharedDataStream<T>::BufferLayout::calculateAndCacheConstants(size_t wordSize, size_t maxReaders) {
709  auto buffer = reinterpret_cast<uint8_t*>(m_buffer->data());
710  m_readerEnabledArray = reinterpret_cast<AtomicBool*>(buffer + calculateReaderEnabledArrayOffset());
711  m_readerCursorArray = reinterpret_cast<AtomicIndex*>(buffer + calculateReaderCursorArrayOffset(maxReaders));
712  m_readerCloseIndexArray = reinterpret_cast<AtomicIndex*>(buffer + calculateReaderCloseIndexArrayOffset(maxReaders));
713  m_dataSize = (m_buffer->size() - calculateDataOffset(wordSize, maxReaders)) / wordSize;
714  m_data = buffer + calculateDataOffset(wordSize, maxReaders);
715 }
