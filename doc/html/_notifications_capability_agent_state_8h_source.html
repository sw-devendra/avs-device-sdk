26 enum class NotificationsCapabilityAgentState {
27  // The capability agent is awaiting directives.
28  IDLE,
29  // A call to renderNotification() has been made, but the notification has not yet finished rendering.
30  PLAYING,
31  // A call to cancelNotificationRendering() has been made, but the notification has not yet been cancelled.
32  CANCELING_PLAY,
33  // The capability agent is shutting down.
34  SHUTTING_DOWN,
35  // The capability agent has finished shutting down.
36  SHUTDOWN
37 };

39 inline std::string stateToString(NotificationsCapabilityAgentState state) {
40  switch (state) {
41  case NotificationsCapabilityAgentState::IDLE:
42  return "IDLE";
43  case NotificationsCapabilityAgentState::PLAYING:
44  return "PLAYING";
45  case NotificationsCapabilityAgentState::CANCELING_PLAY:
46  return "CANCELING_PLAY";
47  case NotificationsCapabilityAgentState::SHUTTING_DOWN:
48  return "SHUTTING_DOWN";
49  case NotificationsCapabilityAgentState::SHUTDOWN:
50  return "SHUTDOWN";
51  }
52  return "Unknown NotificationsCapabilityAgent State";
53 }
