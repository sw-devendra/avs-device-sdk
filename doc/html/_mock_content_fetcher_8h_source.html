225 class MockContentFetcher : public avsCommon::sdkInterfaces::HTTPContentFetcherInterface {
226 public:
227  MockContentFetcher(const std::string& url) : m_url{url} {
228  }
229 
230  std::unique_ptr<avsCommon::utils::HTTPContent> getContent(
231  FetchOptions fetchOption,
232  std::shared_ptr<avsCommon::avs::attachment::AttachmentWriter> writer,
233  const std::vector<std::string>& customHeaders = std::vector<std::string>()) {
234  if (fetchOption == FetchOptions::CONTENT_TYPE) {
235  auto it1 = urlsToContentTypes.find(m_url);
236  if (it1 == urlsToContentTypes.end()) {
237  return nullptr;
238  } else {
239  std::promise<long> statusPromise;
240  auto statusFuture = statusPromise.get_future();
241  statusPromise.set_value(HTTP_REQUEST_OK);
242  std::promise<std::string> contentTypePromise;
243  auto contentTypeFuture = contentTypePromise.get_future();
244  contentTypePromise.set_value(it1->second);
245  return avsCommon::utils::memory::make_unique<avsCommon::utils::HTTPContent>(
246  std::move(statusFuture), std::move(contentTypeFuture), nullptr);
247  }
248  } else if (fetchOption == FetchOptions::ENTIRE_BODY) {
249  auto it2 = urlsToContent.find(m_url);
250  if (it2 == urlsToContent.end()) {
251  return nullptr;
252  } else {
253  static bool liveStreamPlaylistRequested = false;
254  if (m_url == TEST_HLS_LIVE_STREAM_PLAYLIST_URL) {
255  if (!liveStreamPlaylistRequested) {
256  it2->second = TEST_HLS_LIVE_STREAM_PLAYLIST_CONTENT_1;
257  liveStreamPlaylistRequested = true;
258  } else {
259  it2->second = TEST_HLS_LIVE_STREAM_PLAYLIST_CONTENT_2;
260  }
261  }
262  std::promise<long> statusPromise;
263  auto statusFuture = statusPromise.get_future();
264  statusPromise.set_value(HTTP_REQUEST_OK);
265  std::promise<std::string> contentTypePromise;
266  auto contentTypeFuture = contentTypePromise.get_future();
267  contentTypePromise.set_value("");
268  return avsCommon::utils::memory::make_unique<avsCommon::utils::HTTPContent>(
269  std::move(statusFuture), std::move(contentTypeFuture), writeStringIntoAttachment(it2->second));
270  }
271  } else {
272  return nullptr;
273  }
274  }
275 
276 private:
277  std::shared_ptr<avsCommon::avs::attachment::InProcessAttachment> writeStringIntoAttachment(
278  const std::string& string) {
279  static int id = 0;
280  std::shared_ptr<avsCommon::avs::attachment::InProcessAttachment> stream =
281  std::make_shared<avsCommon::avs::attachment::InProcessAttachment>(std::to_string(id++));
282  if (!stream) {
283  return nullptr;
284  }
285  auto writer = stream->createWriter();
286  if (!writer) {
287  return nullptr;
288  }
289  avsCommon::avs::attachment::AttachmentWriter::WriteStatus writeStatus;
290  writer->write(string.data(), string.size(), &writeStatus);
291  return stream;
292  };
293 
294  std::string m_url;
295 };
