229 template <typename Rep, typename Period, typename Task, typename... Args>
230 bool Timer::start(
231  const std::chrono::duration<Rep, Period>& delay,
232  const std::chrono::duration<Rep, Period>& period,
233  PeriodType periodType,
234  size_t maxCount,
235  Task task,
236  Args&&... args) {
237  if (delay < std::chrono::duration<Rep, Period>::zero()) {
238  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "negativeDelay"));
239  return false;
240  }
241  if (period < std::chrono::duration<Rep, Period>::zero()) {
242  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "negativePeriod"));
243  return false;
244  }
245 
246  // Can't start if already running.
247  if (!activate()) {
248  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "timerAlreadyActive"));
249  return false;
250  }
251 
252  // Join old timer thread (if any).
253  if (m_thread.joinable()) {
254  m_thread.join();
255  }
256 
257  // Remove arguments from the task's type by binding the arguments to the task.
258  using BoundTaskType = decltype(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
259  auto boundTask = std::make_shared<BoundTaskType>(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
260 
261  // Remove the return type from the task by wrapping it in a lambda with no return value.
262  auto translatedTask = [boundTask]() { boundTask->operator()(); };
263 
264  // Kick off the new timer thread.
265  m_thread = std::thread{
266  std::bind(&Timer::callTask<Rep, Period>, this, delay, period, periodType, maxCount, translatedTask)};
267 
268  return true;

281 template <typename Rep, typename Period, typename Task, typename... Args>
282 auto Timer::start(const std::chrono::duration<Rep, Period>& delay, Task task, Args&&... args)
283  -> std::future<decltype(task(args...))> {
284  // Can't start if already running.
285  if (!activate()) {
286  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "timerAlreadyActive"));
287  using FutureType = decltype(task(args...));
288  return std::future<FutureType>();
289  }
290 
291  // Join old timer thread (if any).
292  if (m_thread.joinable()) {
293  m_thread.join();
294  }
295 
296  // Remove arguments from the task's type by binding the arguments to the task.
297  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
298 
299  /*
300  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
301  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
302  * will then return a future of the correct type.
303  */
304  using PackagedTaskType = std::packaged_task<decltype(boundTask())()>;
305  auto packagedTask = std::make_shared<PackagedTaskType>(boundTask);
306 
307  // Remove the return type from the task by wrapping it in a lambda with no return value.
308  auto translatedTask = [packagedTask]() { packagedTask->operator()(); };
309 
310  // Kick off the new timer thread.
311  static const size_t once = 1;
312  m_thread = std::thread{
313  std::bind(&Timer::callTask<Rep, Period>, this, delay, delay, PeriodType::ABSOLUTE, once, translatedTask)};
314 
315  return packagedTask->get_future();
316 }

318 template <typename Rep, typename Period>
319 void Timer::callTask(
320  std::chrono::duration<Rep, Period> delay,
321  std::chrono::duration<Rep, Period> period,
322  PeriodType periodType,
323  size_t maxCount,
324  std::function<void()> task) {
325  // Timepoint to measure delay/period against.
326  auto now = std::chrono::steady_clock::now();
327 
328  // Flag indicating whether we've drifted off schedule.
329  bool offSchedule = false;
330 
331  for (size_t count = 0; maxCount == FOREVER || count < maxCount; ++count) {
332  auto waitTime = (0 == count) ? delay : period;
333  {
334  std::unique_lock<std::mutex> lock(m_waitMutex);
335 
336  // Wait for stop() or a delay/period to elapse.
337  if (m_waitCondition.wait_until(lock, now + waitTime, [this]() { return m_stopping; })) {
338  m_stopping = false;
339  m_running = false;
340  return;
341  }
342  }
343 
344  switch (periodType) {
345  case PeriodType::ABSOLUTE:
346  // Update our estimate of where we should be after the delay.
347  now += waitTime;
348 
349  // Run the task if we're still on schedule.
350  if (!offSchedule) {
351  task();
352  }
353 
354  // If the task runtime put us off schedule, skip the next task run.
355  if (now + period < std::chrono::steady_clock::now()) {
356  offSchedule = true;
357  } else {
358  offSchedule = false;
359  }
360  break;
361 
362  case PeriodType::RELATIVE:
363  task();
364  now = std::chrono::steady_clock::now();
365  break;
366  }
367  }
368  m_stopping = false;
369  m_running = false;
370 }
