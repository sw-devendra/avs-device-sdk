320 template <typename ParamType>
321 bool CurlEasyHandleWrapper::setopt(CURLoption option, ParamType value) {
322  auto result = curl_easy_setopt(m_handle, option, value);
323  if (result != CURLE_OK) {
324  logger::acsdkError(logger::LogEntry("CurlEasyHandleWrapper", "setoptFailed")
325  .d("reason", "curl_easy_setopt failed")
326  .d("option", option)
327  .sensitive("value", value)
328  .d("result", result)
329  .d("error", curl_easy_strerror(result)));
330  return false;
331  }
332  return true;
333 }
